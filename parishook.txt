-- Hidden Keys
local function d(k)
    return k:reverse():sub(4, #k-3):gsub("%d", function(c) return string.char(tonumber(c) + 65) end)
end

    local GUI_Enabled = false

    local ESP_Enabled = false
    local SilentAim_Enabled = false
    local Manipulation_Enabled = false

    local keybinds = {
        UI = Enum.KeyCode.LeftControl,
        ESP = Enum.KeyCode.E,
        SilentAim = Enum.KeyCode.F,
        Manipulation = Enum.KeyCode.M
    }

    ScreenGui.Name = "ParisHook"
    ScreenGui.ResetOnSpawn = false
    ScreenGui.Enabled = false
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")

    local MainFrame = Instance.new("Frame")
    MainFrame.Size = UDim2.new(0, 460, 0, 380)

    MainFrame.BackgroundColor3 = Color3.fromRGB(18, 18, 22)

    MainFrame.Parent = ScreenGui

    local Title = Instance.new("TextLabel")
    Title.Size = UDim2.new(1, 0, 0, 45)
    Title.BackgroundColor3 = Accent
    Title.Text = "PARIS.HOOK"
    Title.TextColor3 = Color3.new(1,1,1)
    Title.TextScaled = true
    Title.Font = Enum.Font.GothamBlack
    Title.Parent = MainFrame

    keyList.Position = UDim2.new(0, 10, 0.5, -110)
    keyList.BackgroundColor3 = Color3.fromRGB(22, 22, 28)
    keyList.BorderSizePixel = 0
    keyList.Active = true
    keyList.Draggable = true

    listTitle.Font = Enum.Font.GothamBold
    listTitle.Parent = keyList

    -- Keybind List Content (update dynamically)
    local function updateKeyList()
        -- Clear and rebuild list
    end

    -- Toggle Logic & Keybinds
    UserInputService.InputBegan:Connect(function(input)
        if input.KeyCode == keybinds.UI then
            GUI_Enabled = not GUI_Enabled
            ScreenGui.Enabled = GUI_Enabled
        elseif input.KeyCode == keybinds.ESP then
            ESP_Enabled = not ESP_Enabled
        elseif input.KeyCode == keybinds.SilentAim then
            SilentAim_Enabled = not SilentAim_Enabled
        elseif input.KeyCode == keybinds.Manipulation then
            Manipulation_Enabled = not Manipulation_Enabled
        end
    end)

    print("✅ Paris.Hook Loaded | Use keybinds")
end
